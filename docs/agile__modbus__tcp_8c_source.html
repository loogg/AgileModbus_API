46static int agile_modbus_tcp_build_request_basis(agile_modbus_t *ctx, int function,
47 int addr, int nb,
48 uint8_t *req)
49{
50 agile_modbus_tcp_t *ctx_tcp = ctx->backend_data;
51
52 /* Increase transaction ID */
53 if (ctx_tcp->t_id < UINT16_MAX)
54 ctx_tcp->t_id++;
55 else
56 ctx_tcp->t_id = 0;
57 req[0] = ctx_tcp->t_id >> 8;
58 req[1] = ctx_tcp->t_id & 0x00ff;
59
60 /* Protocol Modbus */
61 req[2] = 0;
62 req[3] = 0;
63
64 /* Length will be defined later by set_req_length_tcp at offsets 4
65 and 5 */
66
67 req[6] = ctx->slave;
68 req[7] = function;
69 req[8] = addr >> 8;
70 req[9] = addr & 0x00ff;
71 req[10] = nb >> 8;
72 req[11] = nb & 0x00ff;
73
74 return AGILE_MODBUS_TCP_PRESET_REQ_LENGTH;
75}

83static int agile_modbus_tcp_build_response_basis(agile_modbus_sft_t *sft, uint8_t *rsp)
84{
85 /* Extract from MODBUS Messaging on TCP/IP Implementation
86 Guide V1.0b (page 23/46):
87 The transaction identifier is used to associate the future
88 response with the request. */
89 rsp[0] = sft->t_id >> 8;
90 rsp[1] = sft->t_id & 0x00ff;
91
92 /* Protocol Modbus */
93 rsp[2] = 0;
94 rsp[3] = 0;
95
96 /* Length will be set later by send_msg (4 and 5) */
97
98 /* The slave ID is copied from the indication */
99 rsp[6] = sft->slave;
100 rsp[7] = sft->function;
101
102 return AGILE_MODBUS_TCP_PRESET_RSP_LENGTH;
103}

122static int agile_modbus_tcp_send_msg_pre(uint8_t *req, int req_length)
123{
124 /* Substract the header length to the message length */
125 int mbap_length = req_length - 6;
126
127 req[4] = mbap_length >> 8;
128 req[5] = mbap_length & 0x00FF;
129
130 return req_length;
131}

153static int agile_modbus_tcp_pre_check_confirmation(agile_modbus_t *ctx, const uint8_t *req,
154 const uint8_t *rsp, int rsp_length)
155{
156 /* Check transaction ID */
157 if (req[0] != rsp[0] || req[1] != rsp[1])
158 return -1;
159
160 /* Check protocol ID */
161 if (rsp[2] != 0x0 && rsp[3] != 0x0)
162 return -1;
163
164 return 0;
165}
