97static uint16_t agile_modbus_rtu_crc16(uint8_t *buffer, uint16_t buffer_length)
98{
99 uint8_t crc_hi = 0xFF; /* high CRC byte initialized */
100 uint8_t crc_lo = 0xFF; /* low CRC byte initialized */
101 unsigned int i; /* will index into CRC lookup */
102
103 /* pass through message buffer */
104 while (buffer_length--) {
105 i = crc_hi ^ *buffer++; /* calculate the CRC */
106 crc_hi = crc_lo ^ _table_crc_hi[i];
107 crc_lo = _table_crc_lo[i];
108 }
109
110 return (crc_hi << 8 | crc_lo);
111}

134static int agile_modbus_rtu_build_request_basis(agile_modbus_t *ctx, int function,
135 int addr, int nb,
136 uint8_t *req)
137{
138 req[0] = ctx->slave;
139 req[1] = function;
140 req[2] = addr >> 8;
141 req[3] = addr & 0x00ff;
142 req[4] = nb >> 8;
143 req[5] = nb & 0x00ff;
144
145 return AGILE_MODBUS_RTU_PRESET_REQ_LENGTH;
146}

199static int agile_modbus_rtu_check_integrity(agile_modbus_t *ctx, uint8_t *msg, const int msg_length)
200{
201 uint16_t crc_calculated;
202 uint16_t crc_received;
203
204 crc_calculated = agile_modbus_rtu_crc16(msg, msg_length - 2);
205 crc_received = (msg[msg_length - 2] << 8) | msg[msg_length - 1];
206
207 /* Check CRC of msg */
208 if (crc_calculated == crc_received)
209 return msg_length;
210
211 return -1;
212}

223static int agile_modbus_rtu_pre_check_confirmation(agile_modbus_t *ctx, const uint8_t *req,
224 const uint8_t *rsp, int rsp_length)
225{
226 /* Check responding slave is the slave we requested (except for broacast
